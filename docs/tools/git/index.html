    多账号配置 # 

    mkdir -p ~/.ssh && cd ~/.ssh
ssh-keygen -t rsa -C cmp@abc.com -f ~/.ssh/github_id_rsa
ssh-keygen -t rsa -C me@abc.com -f ~/.ssh/github_id_rsa_me
ssh-add -D
ssh-add -l
# 配置公钥

    vi config

    # cmp
Host github.com
HostName github.com
PreferredAuthentications publickey
IdentityFile ~/.ssh/github_id_rsa
User ademain

# me
Host github.com-me
HostName github.com
PreferredAuthentications publickey
IdentityFile ~/.ssh/github_id_rsa_me
User ademain

    测试

    ssh -T git@github.com-me
