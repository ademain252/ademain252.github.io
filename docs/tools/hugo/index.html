    #

    搭建 # 

    brew install hugo && hugo new site bonvoyage
cd bonvoyage && git init 
git submodule add https://github.com/alex-shpak/hugo-book themes/hugo-book
echo theme = \'hugo-book\' >> config.toml
hugo new docs/hugo.md
hugo server -D

    配置 #

    vim config.toml

    baseURL = "ademain252.github.io"
languageCode = "zh-CN"
title = "A demain"                     
theme = "hugo-book"

    配置Github # 

    # create a new repository with name `ademain252.github.io`, change config.toml baseURL to `ademain252.github.io`
hugo -D
cd public
git init
git config user.name "ademain" && git config user.email "ademain0612@gmail.com"
git add . && git commit -a -m "first commit"
git log --pretty="%H %an <%ae>"
git remote add origin git@github.com-hugo:ademain252/ademain252.github.io.git
cat .git/config
git push -u origin master
