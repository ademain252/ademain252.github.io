    #

    搭建 # 

    brew install hugo && hugo new site bonvoyage
cd bonvoyage && git init 
git submodule add https://github.com/alex-shpak/hugo-book themes/hugo-book
echo theme = \'hugo-book\' >> config.toml
hugo new docs/hugo.md
hugo server -D

    配置 #

    vim config.toml

    baseURL = "me.github.io"
languageCode = "zh-CN"
title = "A demain"                     
theme = "hugo-book"

    配置Github # 

    # create a new repository with name `me.github.io`, change config.toml baseURL to `me.github.io`
